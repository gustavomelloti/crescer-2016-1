--3
Select NomeEmpregado, DATEDIFF(MONTH, DataAdmissao, '2000-12-31') as QuantidadeMesesTrabalhados
From Empregado
Where DataAdmissao between '1980-05-01' and '1982-01-20'

--4
Select TOP 1  cargo
From Empregado
group by (cargo)
Order by (count(1)) desc
